	def amount: int = 1000;
	def current_value: int = 2;
	while (current_value <= amount) {
		if (current_value % 2 == 0) {
			print(current_value + " is even");
		} else {
			print(current_value + " is odd");
		}
	}
	say_hello("Test", 30);
	return "Done!";
}

route say_hello(name : string, age: int) : nothing {
	def born_in: int = 2021 - age;
	print("Hi ", name, ", you were born in: ", born_in);
}
